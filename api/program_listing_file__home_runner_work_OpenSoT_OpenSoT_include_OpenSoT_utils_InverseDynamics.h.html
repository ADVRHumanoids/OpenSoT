public:

    const std::vector<AffineHelper>& getContactsWrenchAffine() const;

    AffineHelper    _qddot;
    std::vector<AffineHelper> _contacts_wrench;
    std::vector<std::string> _links_in_contact;
    XBot::ModelInterface& _model;
